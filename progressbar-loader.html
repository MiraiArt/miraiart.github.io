Progress bar / Loader

Progress bar

Show head progress bar
  • HTML
  • JS
                
                    <!-- Progress bar -->
                    <div class="progress-bar">
                        <div class="progress bg-blue" style="width: 15%;"></div>
                    </div>
                    
                    <!-- Head Progress bar -->
                    <div class="head-progress">
                        <div class="progress-bar">
                            <div class="progress bg-blue" style="width: 100%;"></div>
                        </div>
                    </div>
                
            
                
                    //Head progress bar animation
                    $('.head-progress .progress-bar').MA_Progress({
                        transitionDuration: 10000,
                        startAt: '0%',
                        endAt: '100%',
                        onStart: function () { console.log(this) },
                        onEnd: function () { console.log(this) },
                        hideAtEnd: true
                    });
