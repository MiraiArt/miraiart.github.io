List style

List style can be used for display list of items

It's possible to add bg-* or border-* class on parent and item for change color.

HTML
                

                    <!-- Default List style -->
                    <ul class="list">
                        <li class="list-item">Item 1</li>
                        <li class="list-item">Item 2</li>
                        <li class="list-item">Item 3</li>
                        <li class="list-item">Item 4</li>
                    </ul>

                    <!-- List style full background color -->
                    <ul class="list bg-pink">
                        <li class="list-item">Item 1</li>
                        <li class="list-item">Item 2</li>
                        <li class="list-item">Item 3</li>
                        <li class="list-item">Item 4</li>
                    </ul>
                    
                    <!-- Background color on item of list -->
                    <ul class="list">
                        <li class="list-item bg-blue">Item 1</li>
                        <li class="list-item">Item 2</li>
                        <li class="list-item bg-blue">Item 3</li>
                        <li class="list-item">Item 4</li>
                    </ul>
                    
                    <!-- List style full border color -->
                    <ul class="list border-green">
                        <li class="list-item">Item 1</li>
                        <li class="list-item">Item 2</li>
                        <li class="list-item">Item 3</li>
                        <li class="list-item">Item 4</li>
                    </ul>
                    
                    <!-- Border color on item of list -->
                    <ul class="list">
                        <li class="list-item border-red">Item 1</li>
                        <li class="list-item">Item 2</li>
                        <li class="list-item border-red">Item 3</li>
                        <li class="list-item">Item 4</li>
                    </ul>
