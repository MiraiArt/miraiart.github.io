                        message: 'Please defined a value !',

                        classHead: 'bg-purple',
                        backgroundClose: false,
                        closeBtn: true,
                        closeIcon: '<i class="fab fa-times"></i>',

                            btnClose: {
                                class: 'bg-blue',
                                name:'dqzdq',
                                action: function () {
                                }
                            },
                            btnRemove: {
                                class: 'bg-red',
                                action: function () {
                                }
                            }
                        }
                    });

Options

Name Type Default value Description
transitionDuration Number 500 It's transition duration for the alert when it's open or close, in milliseconds.
title String null It's alert title.
message String null It's message display in alert.
